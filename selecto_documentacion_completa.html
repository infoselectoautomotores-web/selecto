Módulo: Ganancias — Fórmulas Definitivas

🚨 CRÍTICO — NO MODIFICAR JAMÁS: La permuta es una forma de cobro. NO suma ni resta de la ganancia.

// FÓRMULAS DEFINITIVAS E INAMOVIBLES

CostoTotal    = Valor_Compra + Σ_Gastos
Ganancia      = Valor_Venta - CostoTotal - Comisión
ROI           = (Ganancia / CostoTotal) × 100
DineroLiquidar = Valor_Venta - Valor_Permuta   ← SOLO informativo

// Ejemplo de validación:
// Compra: $8.000.000 | Gastos: $500.000 | Venta: $10.000.000
// Permuta: $2.000.000 | Comisión: $0
//
// CostoTotal    = 8.000.000 + 500.000    = $8.500.000
// Ganancia      = 10.000.000 - 8.500.000 = $1.500.000 ✅
// ROI           = 1.500.000 / 8.500.000  = 17.65% ✅
// DineroLiquidar = 10.000.000 - 2.000.000 = $8.000.000 (informativo)

Distribución por socios

pct_socio  = capital_socio / CostoTotal
gan_socio  = Ganancia × pct_socio
gastos_soc = Σ gastos donde pagador = nombre_socio
total_cobrar = capital_socio + gastos_soc + gan_socio

// onAfterSave: módulo Autosave — debounce 600ms
function onAfterSave() {
  _isDirty = true;
  showSavingIndicator();          // "Guardando..." (dorado parpadeante)
  clearTimeout(_saveTimer);
  _saveTimer = setTimeout(() => {
    _lastSavedAt = new Date();
    _isDirty = false;
    showSavedIndicator();          // "🟢 Datos guardados" con hora
    checkAutoBackup();             // Si >24h sin backup → createBackupNow(true)
  }, 600);
}

function trend(curr, prev) {
  if (!prev || prev === 0) return { icon: '',   color: 'var(--tx3)' };
  if (curr > prev)         return { icon: ' ▲', color: 'var(--green)' };
  if (curr < prev)         return { icon: ' ▼', color: 'var(--red)' };
  return                          { icon: ' —', color: 'var(--tx3)' };
}
